package com.connsulted.aomi

import android.content.Context
import android.util.Log
import java.io.File
import java.io.IOException

class AomiOrchestrator(private val context: Context) {

    private val TAG = "AomiOrchestrator"
    private var llamaProcess: Process? = null
    private var ollaProcess: Process? = null

    fun initializeServices(localModelPath: String, pcNetworkIp: String) {
        // 1. Start the local Llama.cpp Server
        startLlamaBinary(localModelPath, "8081")

        // 2. Configure Olla to route exclusively to the local Llama server,
        // with a failover to your PC's LM Studio instance if the local model crashes.
        val ollaConfigJson = """
        {
          "server": { "port": 8080 },
          "routing": { "strategy": "fallback" },
          "endpoints": [
            {
              "name": "Local-Llama-CPP",
              "url": "http://127.0.0.1:8081",
              "provider": "llamacpp"
            },
            {
              "name": "PC-Failover",
              "url": "http://$pcNetworkIp:1234",
              "provider": "openai"
            }
          ]
        }
        """

        // 3. Start Olla Proxy
        startOllaBinary(ollaConfigJson)
    }

    private fun startLlamaBinary(modelPath: String, port: String) {
        if (llamaProcess != null) return

        try {
            val nativeLibraryDir = context.applicationInfo.nativeLibraryDir
            val llamaExecutable = File(nativeLibraryDir, "libllama-server.so")

            if (!llamaExecutable.exists()) {
                Log.e(TAG, "Llama binary not found!")
                return
            }
            llamaExecutable.setExecutable(true)

            // Llama server arguments
            val processBuilder = ProcessBuilder(
                llamaExecutable.absolutePath,
                "-m", modelPath,
                "-c", "2048",
                "--port", port
            )
            processBuilder.redirectErrorStream(true)

            llamaProcess = processBuilder.start()
            Log.i(TAG, "Llama Server started. PID: ${llamaProcess?.hashCode()}")

            // Consume Llama logs
            Thread {
                llamaProcess?.inputStream?.bufferedReader()?.useLines { lines ->
                    lines.forEach { Log.d("Llama-Microservice", it) }
                }
            }.start()

        } catch (e: Exception) {
            Log.e(TAG, "Failed to start Llama binary", e)
        }
    }

    private fun startOllaBinary(configJson: String) {
        if (ollaProcess != null) return

        try {
            val configDir = File(context.filesDir, "olla_data")
            if (!configDir.exists()) configDir.mkdirs()
            
            val configFile = File(configDir, "config.json")
            configFile.writeText(configJson)

            val nativeLibraryDir = context.applicationInfo.nativeLibraryDir
            val ollaExecutable = File(nativeLibraryDir, "libolla.so")

            if (!ollaExecutable.exists()) {
                Log.e(TAG, "Olla binary not found!")
                return
            }
            ollaExecutable.setExecutable(true)

            val processBuilder = ProcessBuilder(
                ollaExecutable.absolutePath,
                "--config", configFile.absolutePath 
            )
            processBuilder.directory(configDir)
            processBuilder.redirectErrorStream(true)

            ollaProcess = processBuilder.start()
            Log.i(TAG, "Olla Proxy started. PID: ${ollaProcess?.hashCode()}")

            // Consume Olla logs
            Thread {
                ollaProcess?.inputStream?.bufferedReader()?.useLines { lines ->
                    lines.forEach { Log.d("Olla-Microservice", it) }
                }
            }.start()

        } catch (e: Exception) {
            Log.e(TAG, "Failed to start Olla binary", e)
        }
    }

    fun shutdown() {
        Log.i(TAG, "Shutting down Aomi microservices...")
        ollaProcess?.destroy()
        llamaProcess?.destroy()
        ollaProcess = null
        llamaProcess = null
    }
}
